\recipe{Roasted Brussel Sprout Salad}

\recipecolumn{
	\begin{itemize}
		\item Brussel Sprouts
		\item Spinach Greens
    \item Balsamic Vinegar
		\item Extra Virgin Olive Oil
    \item Honey
    \item Walnuts
    \item Goats Cheese
    \item Cranberries
	\end{itemize}
}{
	\begin{enumerate}
		\item In a small saucepan, bring balsamic vinegar and honey together with a little water till reduced by a quarter and can still be used as a dressing
    \item Cut brussel sprouts and roast in the oven with olive oil, salt, and pepper at 400 degrees. Let brussel sprouts cool to room temperatures and chop into smaller pieces.
		\item Mix together spinach, brusse sprouts, walnuts and cranberries in a bowl. Add goats cheese on top and toss with balasamic vinegar dressing and olive oil.
	\end{enumerate}
}

\recipe{Turmeric Roasted Potato Salad}
\recipecolumn{
	\begin{enumerate}
		\item $\frac{1}{2}$ lb red potatoes (quartered);
		\item Paprika;
		\item Salt;
		\item Black Pepper;
		\item Cumin;
		\item Honey;
		\item Parsley;
	\end{enumerate}
}{
	\begin{itemize}
		\item Preheat the oven to 425. Toss potatoes in olive oil, paprika, salt, pepper, and cumin. Place potatoes on a tray and roast for 20 minutes
		\item In a separate bowl, mix together greek yogurt, turmeric, honey, cumin and parsley to make dressing for the potatoes. Season as necessary with salt, pepper and olive oil.
		\item After taking out roasted potatoes, let cool for 5 minutes and then toss in bowl with dressing. Garnish with extra parsley if needed.\footnote{You can mash up the potatoes after a day and form into patties for a makeshift pav bhaji}
	\end{itemize}
}
