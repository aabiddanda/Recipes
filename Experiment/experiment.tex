\recipe{Eggs in a Basket}
\ingred{Slice of Bread; 
Egg;
Butter;
Black Pepper;
Green Onion (finely chopped)
}

\begin{algorithmic}[0]
\Procedure{Egg-in-a-Basket}{$Egg,Bread,Butter$}
   \State $Bread \gets Make\; Hole\; in\; Bread$
   \State $Pan \gets Bread$
   \State $Pan \gets Crack\;egg\;and\;put\;in\;Bread$
   \Repeat
	   \State $Time = 0\; min$
	   \While{$Time \neq 3\; min$}
	   		\State $Time \gets Time + 1\;min$
	   \EndWhile\label{euclidendwhile}
	   \State $Bread \gets Flipped$
	 \Until{$Egg\;is\;Cooked$}
	 \State{$Bread \gets Pepper + Green\;Onion$}
\EndProcedure
\end{algorithmic}
