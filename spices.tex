
\recipe{Moroccan Meat Spice}\footnote{if using all dried spices, toast spices for max of 1 minute on low-medium heat.}
\recipecolumn{
	\begin{itemize}
		\item Brown Sugar (1 Tbsp)
		\item Parsley (1/2 Tbsp)
		\item Cinnamon (1/2 Tbsp)
		\item Cumin (1 Tbsp)
		\item Coriander (1 Tbsp)
		\item Garlic Powder (1/2 Tbsp)
		\item Ginger Powder (1/2 Tbsp)
		\item Salt (1 Tbsp)
		\item Black Pepper (1 Tbsp)
		\item Red Chili Flakes (1/4 Tbsp)\footnote{Adjust this to taste for heat vs. sweet}
	\end{itemize}
}{
	\begin{enumerate}
	\item Toast cinnamon, cumin, coriander, garlic powder, and ginger powder in a pan.
	\item Place toasted spices and mix with brown sugar, parsley, salt, black pepper, and chili flakes in a jar and mix. 
	\item Use as dry rub for steaks or mix into ground meat for burgers (see Morroccan Turkey Burgers)
	\end{enumerate}
}
