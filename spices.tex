
\recipe{Moroccan Meat Spice}
\recipecolumn{
	\begin{itemize}
		\item Brown Sugar
		\item Parsley
		\item Cinnamon
		\item Cumin
		\item Garlic Powder 
		\item Ginger Powder
	\end{itemize}
}{
	\begin{enumerate}
	\item Toast cinnamon, cumin, garlic powder, and ginger powder in a pan.
	\item Place toasted spices and mix with brown sugar and parsley in jar
	\item Use as dry rub for steaks or mix into ground meat for burgers (see Morroccan Turkey Burgers)
	\end{enumerate}
}

\recipe{Soy Ginger Marinade}
\recipecolumn{
	\begin{itemize}
		\item Honey
		\item Soy Sauce
		\item Ginger (chopped) 
		\item Garlic (minced)
	\end{itemize}
}{
	\begin{enumerate}
		\item Mix all ingredients together in bowl and pour over meat or veggies of choice.
		\item \textbf{Optional :} In pan, lightly fry ginger and garlic in small amount of oil and add oil to marinade as well.
	\end{enumerate}
}
